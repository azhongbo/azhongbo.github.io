# Perl 兩個日期相減
use POSIX qw(mktime);
use POSIX qw(difftime);
$kk = &diff_date("2014/11/12","2014/11/13");
sub diff_date()
    {
        my @t1 = split( "/" , $_[0] );
        $year = @t1[0] - 1900;
        $mon  = @t1[1] - 1;
        $mday = @t1[2];
        $sec  = 00;
        $min  = 00;
        $hour = 00;
        my $start = mktime ($sec,$min,$hour,$mday,$mon,$year);
        my @t2 = split( "/" , $_[1] );
        $year = @t2[0] - 1900;
        $mon  = @t2[1] - 1;
        $mday = @t2[2];
        $sec  = 00;
        $min  = 00;
        $hour = 00;
        my $end = mktime ($sec,$min,$hour,$mday,$mon,$year);
        my $result = difftime($end, $start);
        print $result/(24*60*60),"\n";
}

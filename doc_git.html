git 基本指令

git init
git add filename.txt
git commit -m "first commit"
git remote add origin https://github.com/azhongbo/phantomjs.git
git push -u origin master


git config --global credential.helper cache
git config --global credential.helper "cache --timeout=3600"


git push -u origin master

## 增加一個連結
git remote add origin https://github.com/azhongbo/phantomjs.git

## 更新一個連結
git remote set-url origin https://github.com/azhongbo/phantomjs.git

## 建立一個連結分支
git checkout -b "test"

## 切換分支
git checkout "test"


## 檢查目前位置
git branch

git push origin "test"

git clone https://github.com/azhongbo/phantomjs.git
